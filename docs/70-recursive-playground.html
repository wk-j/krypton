export function buildFeatureSettings(features: string[]): string { if (features.length === 0) return ''; const parts: string[] = []; for (const raw of features) { if (raw.startsWith('-')) { const tag = raw.slice(1); parts.push(`"${tag}" 0`); } else { const tag = raw.startsWith('+') ? raw.slice(1) : raw; parts.push(`"${tag}" 1`); } } return parts.join(', '); }
